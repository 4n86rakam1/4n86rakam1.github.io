    package main

import (
    "fmt"
    "os"
    "strings"
)

var flag = "irisctf{this_is_not_the_real_flag}"

func init() {
    runed := []string{}
    z := rune(0)

    for _, v := range flag {
        runed = append(runed, string(v+z))
        z = v
    }

    flag = strings.Join(runed, "")
}

func main() {
    file, err := os.OpenFile("the", os.O_RDWR|os.O_CREATE, 0644)
    if err != nil {
        fmt.Println(err)
        return
    }

    defer file.Close()
    if _, err := file.Write([]byte(flag)); err != nil {
        fmt.Println(err)
        return
    }
}

    Solution

    The flag is encrypted by encrypted[i] = flag[i]+flag[i-1]. Thus, the original flag is decrypted by flag[i] = encrypted[i]-flag[i-1] with flag[0] = 'i'

    with open("the") as f:
    decrypted = f.read()

flag = ""
previous = 0
for v in decrypted:
    previous = ord(v) - previous
    flag += chr(previous)

print(flag)

    Or golang version solver:

    package main

import (
    "fmt"
    "os"
    "strings"
)

func main() {
    decrypted, err := os.ReadFile("the")
    if err != nil {
        fmt.Println(err)
        return
    }

    flag := []string{}
    var previous  = rune(0)

    // convert it to a string avoid reading at the bytes level, iterating in the order of 0x69, 0xc3, 0x9b, 0xc3, 0x9b, ...
    // https://stackoverflow.com/a/71716955/10819876
    for _, v := range string(decrypted) {
        previous = v - previous
        flag = append(flag, string(previous))
    }

    fmt.Print(strings.Join(flag, ""))
}
