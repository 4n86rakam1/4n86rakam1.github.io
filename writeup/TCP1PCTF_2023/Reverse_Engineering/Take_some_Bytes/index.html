    Thus, implemented solver.py

    flag = [" " for _ in range(25)]

flag[0:6] = "TCP1P{"
flag[6:10] = "byte"

flag[10] = "_"
flag[15] = "_"
flag[18] = "_"

flag[11:15] = "code"
flag[19] = "c"

flag[20] = "o"

flag[16] = "i"
flag[17] = "s"

flag[19] = "H"

flag[20] = "u"

flag[21] = "F"

flag[22] = "t"
flag[23] = "t"

flag[24] = "}"

print("".join(flag))
